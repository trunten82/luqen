    ```ts
    // packages/<svc>/tests/openapi/route-coverage.test.ts
    import { describe, it, expect } from 'vitest';
    import { buildApp } from '../../src/server.js'; // or buildMcpApp for MCP
    describe('OpenAPI route coverage', () => {
      it('every registered route appears in /docs/json paths', async () => {
        const app = await buildApp();
        await app.ready();
        const spec: any = app.swagger();
        const registeredRoutes = app.printRoutes({ commonPrefix: false })
          .split('\n')
          .map(l => l.trim())
          .filter(Boolean)
          .map(parseRouteLine); // helper: returns { method, path }
        // For MCP test only: filter to routes with path.startsWith('/mcp')
        for (const r of registeredRoutes) {
          if (r.path === '*' || r.path.startsWith('/__')) continue;
          const pathKey = r.path; // OpenAPI uses :param → {param}; convert
          expect(spec.paths, `missing path ${r.method} ${r.path}`).toHaveProperty(toOpenApiPath(pathKey));
        }
        await app.close();
      });
    });
    ```
    Implement `parseRouteLine` and `toOpenApiPath` helpers inline or in a shared `tests/openapi/util.ts`.

    The test MUST FAIL initially if any route is missing a schema (which is what makes Task 2 a real RED→GREEN loop).
  </action>
