    1. Add to the existing delegated change/submit listener in agent.js (the same pattern that handles `retryAssistant`, `editUserMessage`, etc.):

       ```js
       // Handler: agentOrgSwitch (form change on the org select)
       // Triggered by data-action="agentOrgSwitch" on the wrapping form.
       async function handleAgentOrgSwitch(form) {
         const select = form.querySelector('.agent-drawer__org-switcher-select');
         const toast = form.querySelector('[data-role="orgToast"]');
         const orgId = select.value;
         const previousOrgId = select.dataset.previousOrgId || '';
         showOrgToast(toast, T('agent.org.switching'), { state: 'loading' });
         try {
           const res = await fetch('/agent/active-org', {
             method: 'POST',
             credentials: 'same-origin',
             headers: { 'Content-Type': 'application/json', 'X-CSRF-Token': csrfToken() },
             body: JSON.stringify({ orgId }),
           });
           if (!res.ok) {
             const code = res.status === 403 ? 'forbidden' : 'error';
             showOrgToast(toast, T(code === 'forbidden' ? 'agent.org.forbidden' : 'agent.org.error'), { state: 'error' });
             // Roll back the select to previous value
             if (previousOrgId) select.value = previousOrgId;
             return;
           }
           const body = await res.json();
           // Success: force-new-conversation, refresh panel, show toast
           conversationId = null;
           select.dataset.previousOrgId = body.activeOrgId;
           showOrgToast(toast, T('agent.org.switched', { orgName: body.activeOrgName }), { state: 'ok' });
           setTimeout(() =&gt; hideOrgToast(toast), 2000);
           await loadPanel();
           announce(T('agent.org.switched', { orgName: body.activeOrgName }));
         } catch (e) {
           showOrgToast(toast, T('agent.org.error'), { state: 'error' });
           if (previousOrgId) select.value = previousOrgId;
         }
       }
       ```

       Hook it into the existing dispatcher: when a `change` event fires on a form whose closest `[data-action]` resolves to `agentOrgSwitch`, call this handler with the form element.

       `showOrgToast(el, text, opts)` and `hideOrgToast(el)` are tiny new helpers — set `.is-visible`, `.is-error` classes, set `textContent` (NEVER innerHTML — T-36-10 pattern).

       `T(key, vars)` is the existing i18n helper in agent.js (search for the existing function); if it's not present, fall back to a minimal interpolator.

       `csrfToken()` — same helper used by retry/edit handlers in Phase 37-04.

       ```js
       async function autoSwitchOrgIfNeeded(targetOrgId, targetOrgName) {
         const select = document.querySelector('.agent-drawer__org-switcher-select');
         if (!select) return true; // no switcher = non-admin = caller is on the same org by definition
         if (select.value === targetOrgId) return true;
         const res = await fetch('/agent/active-org', {
           method: 'POST',
           credentials: 'same-origin',
           headers: { 'Content-Type': 'application/json', 'X-CSRF-Token': csrfToken() },
           body: JSON.stringify({ orgId: targetOrgId }),
         });
         if (!res.ok) {
           const toast = document.querySelector('[data-role="orgToast"]');
           if (toast) showOrgToast(toast, T('agent.org.error'), { state: 'error' });
           return false;
         }
         select.value = targetOrgId;
         select.dataset.previousOrgId = targetOrgId;
         const toast = document.querySelector('[data-role="orgToast"]');
         if (toast) {
           showOrgToast(toast, T('agent.org.switched', { orgName: targetOrgName }), { state: 'ok' });
           setTimeout(() =&gt; hideOrgToast(toast), 2000);
         }
         return true;
       }
       ```
