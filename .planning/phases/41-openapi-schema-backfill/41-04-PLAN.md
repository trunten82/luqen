import { Type, type TSchema } from '@sinclair/typebox';

export const ErrorEnvelope = Type.Object({
  error: Type.String(),
  statusCode: Type.Optional(Type.Number()),
  message: Type.Optional(Type.String()),
}, { $id: 'ErrorEnvelope', additionalProperties: true });

export const LuqenResponse = <T extends TSchema>(data: T) => Type.Object({
  success: Type.Optional(Type.Boolean()),
  data: Type.Union([data, Type.Null()]),
  error: Type.Optional(Type.Union([Type.String(), Type.Null()])),
  meta: Type.Optional(Type.Object({}, { additionalProperties: true })),
}, { additionalProperties: true });

// Empty 204 response shape (Type.Null() is correct for explicitly empty bodies):
export const NoContent = Type.Null();

// HTML-page hide marker — use as schema option on every Handlebars-rendering route:
//   app.get('/admin/llm', { schema: { hide: true } }, async (req, reply) => reply.view(...))
// Routes flagged hide: true are NOT in the OpenAPI spec and DO NOT need to satisfy the
// route-vs-spec coverage test — that test compares routes vs spec paths, but the
// existing test in 40-01 enumerates ALL routes regardless of hide. We must therefore
// either (a) modify the test to skip hide:true routes, or (b) include them in the spec.
// PREFERRED: keep them in the spec via tags: ['html-page'] + summary, response: 200 = Type.String()
//   so the spec accurately reflects "this route returns text/html". This matches D-05 tolerance.
```
