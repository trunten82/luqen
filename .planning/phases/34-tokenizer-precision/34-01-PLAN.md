       ```typescript
       export interface OllamaTokenizerConfig { readonly baseUrl: string; }

       interface CacheEntry {
         readonly tokenizerModel: string | null;   // from /api/show details.tokenizer
         readonly vocabSize: number | null;
         readonly avgCharsPerToken: number;        // derived; fallback 3.5
       }

       export const LARGE_VOCAB_THRESHOLD = 100000;

       const cache = new Map<string, CacheEntry>();
       const inflight = new Map<string, Promise<void>>();
       let config: OllamaTokenizerConfig = { baseUrl: process.env.OLLAMA_BASE_URL ?? 'http://localhost:11434' };

       export function configureOllamaTokenizer(next: OllamaTokenizerConfig): void {
         // baseUrl MUST be a fixed server-owned config — not user-controlled per-request.
         const trimmed = next.baseUrl.replace(/\/$/, '');
         config = { baseUrl: trimmed };
       }

       export function warm(model: string): Promise<void> {
         if (cache.has(model)) return Promise.resolve();
         // In-flight dedup: concurrent warm(model) calls for the same model share one /api/show request.
         const existing = inflight.get(model);
         if (existing) return existing;
         const p = (async () => {
           try {
             const res = await fetch(\`\${config.baseUrl}/api/show\`, {
               method: 'POST',
               headers: { 'Content-Type': 'application/json' },
               body: JSON.stringify({ name: model }),
             });
             if (!res.ok) return;
             const data = await res.json() as { details?: { tokenizer?: string }; model_info?: Record<string, unknown> };
             const vocabSize = extractVocabSize(data.model_info);
             cache.set(model, {
               tokenizerModel: data.details?.tokenizer ?? null,
               vocabSize,
               avgCharsPerToken: vocabSize !== null && vocabSize > LARGE_VOCAB_THRESHOLD ? 3.2 : 3.5,
             });
           } catch {
             // Silent fail — caller falls back to char/4 per D-03.
           } finally {
             inflight.delete(model);
           }
         })();
         inflight.set(model, p);
         return p;
       }

       export function countText(model: string, text: string): number | undefined {
         const entry = cache.get(model);
         if (!entry) return undefined;
         if (text.length === 0) return 0;
         return Math.ceil(text.length / entry.avgCharsPerToken);
       }

       function extractVocabSize(info: Record<string, unknown> | undefined): number | null {
         if (!info) return null;
         for (const [k, v] of Object.entries(info)) {
           if (k.endsWith('.vocab_size') && typeof v === 'number') return v;
         }
         return null;
       }

       export function _resetOllamaCacheForTest(): void { cache.clear(); }
       ```
