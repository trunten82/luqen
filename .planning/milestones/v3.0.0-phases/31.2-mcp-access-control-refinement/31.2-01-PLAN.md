     if (orgId === 'system') return [];
     const rows = this.db
       .prepare(
         `SELECT DISTINCT c.*
            FROM oauth_clients_v2 c
            JOIN team_members tm ON tm.user_id = c.registered_by_user_id
            JOIN teams t          ON t.id       = tm.team_id
