    ```typescript
    /**
     * registerPrompts — Dashboard MCP Prompts (Phase 30 plan 30-05, MCPI-06).
     *
     * Three chat-message template prompts locked by 29-CONTEXT.md D-12 and
     * 30-CONTEXT.md D-13 through D-15:
     *
     *   /scan    — ask the LLM to scan a site and summarise top issues
     *   /report  — ask the LLM to summarise an existing scan report
     *   /fix     — ask the LLM to generate a code-fix for a specific WCAG issue
     *
     * Each prompt returns ONE user message whose text begins with a tool-aware
     * system preamble ("System: ...\n\nUser: ..."). The MCP 1.27.1 SDK's
     * PromptMessageSchema does NOT include a 'system' role — so we embed the
     * preamble in the user message text instead. Behaviour matches D-15 (tool-
     * aware, not tool-prescriptive) — the LLM, not the prompt, picks which
     * tools to invoke and in what order.
     *
     * D-17 invariant: NO argsSchema on any prompt contains 'orgId'. orgId is
     * sourced from the JWT via the ToolContext ALS at tool-invocation time.
     */

    import { z } from 'zod';
    import type { McpServer } from '@modelcontextprotocol/sdk/server/mcp.js';

    export const DASHBOARD_PROMPT_NAMES = ['scan', 'report', 'fix'] as const;

    /**
     * Tool-aware system preamble embedded at the start of each prompt's user
     * message. Enumerates cross-service tools but does NOT prescribe sequencing
     * (D-15). Listing tools here does NOT imply the caller is authorised to
     * invoke them — permission enforcement happens at tool-invocation via
     * resolveEffectivePermissions (Phase 28 D-03).
     */
    const SYSTEM_PREAMBLE =
      'You are a WCAG compliance assistant in the Luqen dashboard. Available tools across services:\n' +
      '  - dashboard_scan_site (trigger new accessibility scan)\n' +
      '  - dashboard_list_reports, dashboard_get_report, dashboard_query_issues (read scan results)\n' +
      '  - dashboard_list_brand_scores, dashboard_get_brand_score (read brand scores)\n' +
      '  - llm_analyse_report (LLM summary of a report)\n' +
      '  - llm_generate_fix (LLM code-fix for a specific WCAG issue)\n' +
      '  - branding_match, branding_list_guidelines, branding_get_guideline (brand guideline lookups)\n' +
      'Pick appropriate tools when the user asks a question; you are not required to follow any specific sequence. Permission enforcement happens at tool-invocation time.';

    function renderUserMessage(userTask: string): {
      role: 'user';
      content: { type: 'text'; text: string };
    } {
      return {
        role: 'user',
        content: {
          type: 'text',
          text: `System: ${SYSTEM_PREAMBLE}\n\nUser: ${userTask}`,
        },
      };
    }

    export function registerPrompts(server: McpServer): void {
      // ------------------------------------------------------------------
      // /scan  —  trigger an accessibility scan + summary
      // ------------------------------------------------------------------
      server.registerPrompt(
        'scan',
        {
          title: 'Scan a site',
          description: 'Scan a website for WCAG compliance and summarize the top issues.',
          argsSchema: {
            siteUrl: z.string().describe('The website URL to scan'),
            standard: z
              .enum(['WCAG2A', 'WCAG2AA', 'WCAG2AAA'])
              .optional()
              .describe('WCAG level: WCAG2A, WCAG2AA, or WCAG2AAA — defaults to WCAG2AA'),
          },
        },
        // orgId: N/A (global — prompt templates are client-facing; tool calls triggered by the prompt source orgId from JWT via ToolContext ALS at invocation time)
        (args) => {
          const standard = args.standard ?? 'WCAG2AA';
          const userTask = `Scan ${args.siteUrl} for WCAG ${standard} compliance and summarize the top 5 issues.`;
          return { messages: [renderUserMessage(userTask)] };
        },
      );

      // ------------------------------------------------------------------
      // /report  —  summarise an existing scan report
      // ------------------------------------------------------------------
      server.registerPrompt(
        'report',
        {
          title: 'Summarize a scan report',
          description: 'Summarize a scan report with executive-level findings grouped by WCAG principle and severity.',
          argsSchema: {
            scanId: z.string().describe('Scan ID returned from dashboard_scan_site or dashboard_list_reports'),
          },
        },
        // orgId: N/A (global — prompt templates are client-facing; tool calls triggered by the prompt source orgId from JWT via ToolContext ALS at invocation time)
        (args) => {
          const userTask = `Retrieve the scan report for ${args.scanId} and summarise findings grouped by WCAG principle and severity.`;
          return { messages: [renderUserMessage(userTask)] };
        },
      );

      // ------------------------------------------------------------------
      // /fix  —  generate a code-fix suggestion for a WCAG issue
      // ------------------------------------------------------------------
      server.registerPrompt(
        'fix',
        {
          title: 'Generate a fix for an issue',
          description: 'Generate an AI code-fix suggestion for a specific WCAG issue.',
          argsSchema: {
            issueId: z.string().describe('The pa11y issue code, e.g. WCAG2AA.Principle1.Guideline1_1.1_1_1.H37'),
            scanId: z.string().optional().describe('Scan context for the issue'),
          },
        },
        // orgId: N/A (global — prompt templates are client-facing; tool calls triggered by the prompt source orgId from JWT via ToolContext ALS at invocation time)
        (args) => {
          const scanSuffix = args.scanId != null && args.scanId !== '' ? ` in scan ${args.scanId}` : '';
          const userTask = `Generate a code fix for WCAG issue ${args.issueId}${scanSuffix}. Include the exact HTML/CSS change and why it resolves the criterion.`;
          return { messages: [renderUserMessage(userTask)] };
        },
      );
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| MCP client → prompts/get | Untrusted arguments (siteUrl, scanId, issueId) interpolated into user-message text |
| Prompt template → LLM | Rendered message shipped to a downstream LLM that may act on embedded tool listing |
| Prompt enumeration → list-tools inference | System preamble enumerates cross-service tools the LLM *could* call — tempting a caller to assume authorisation |
| Prompt name registry → SDK internals | Prompt names collide with SDK-reserved slots could cause registration failure or override |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-30-05-01 | Tampering | Prompt argument injection — caller supplies `siteUrl` containing shell metacharacters, template strings, or newlines that break the rendered user message | mitigate | Prompts are pure text template renderers. The interpolated string is passed as `{type: 'text', text: ...}` content; it is NEVER executed server-side. Any subsequent tool calls triggered by the LLM will re-validate their own inputs via zod schemas (Phase 28/30 tool handlers). Tests 4-8 exercise the interpolation path; no code path evaluates `args.siteUrl`, `args.scanId`, or `args.issueId`. |
| T-30-05-02 | Information Disclosure | System preamble leaks `ctx.orgId` or other caller-scoped state into the rendered text | mitigate | `SYSTEM_PREAMBLE` is a module-level const string — zero interpolation of `ctx.orgId` or any ToolContext field. The classification comment on each handler is `// orgId: N/A (global — ...)` — prompts NEVER touch ctx. D-17 iteration test (Task 2 test 9) confirms no `orgId` key in argsSchema. Belt-and-braces: grep acceptance criterion `grep -n "ctx" packages/dashboard/src/mcp/prompts.ts` MUST return NO match. |
| T-30-05-03 | Elevation of Privilege | System preamble enumerates tools the caller is NOT authorised to invoke, leading the LLM to attempt calls that 403 and potentially leak tool names via error messages | mitigate | The preamble is intentionally tool-aware but non-prescriptive (D-15). Permission enforcement happens at tool-invocation time via `resolveEffectivePermissions` (Phase 28 D-03). Tool-manifest visibility (tools/list) is the authoritative source of what the LLM can call; the preamble is a hint, not a grant. A caller without `branding.view` will see `branding_*` tools filtered OUT of their tools/list response — even if the prompt mentions them, the LLM cannot invoke them. The 403 response envelope from tool invocation is a generic `isError: true` with a scoped error message, not the underlying permission name. |
| T-30-05-04 | Information Disclosure | Over-permissive tool listing in preamble acts as an advertisement for tools that exist but are not exposed to this caller | accept | The Luqen tool surface is already public-knowledge (documented in ROADMAP + README + MCP reference docs). Enumerating tool names in the preamble leaks no secret. Mitigated by: (a) tools/list is authoritative for what the caller CAN invoke; (b) errors on unauthorised calls are generic; (c) tool names are not sensitive. |
| T-30-05-05 | Spoofing | Prompt name collision with SDK-reserved prefix breaks registration or overrides an SDK-internal prompt | mitigate | The SDK does NOT reserve any prompt names (verified in `node_modules/@modelcontextprotocol/sdk/dist/esm/server/mcp.js` lines 397-446 — there is no built-in prompt registry; `setPromptRequestHandlers` only uses user-registered entries). The names `scan`, `report`, `fix` are simple identifiers — not namespaced (e.g. `mcp.internal.*`) — and are unlikely to collide. If a future SDK release reserves a name, registration would throw at `createDashboardMcpServer()` construction, surfaced immediately by Task 2 test 2 (prompts/list assertion). |
| T-30-05-06 | Elevation of Privilege | A caller without tool-invocation authorisation uses prompts/get as a discovery channel to map available tools | accept | As per T-30-05-03 rationale. The MCP protocol's list-primitives endpoints (tools/list, prompts/list, resources/list) are authenticated but intentionally enumerate the server's surface. Permission enforcement happens at invocation, not discovery. Consistent with Phase 28 posture. |
| T-30-05-07 | Tampering | Future developer adds a 4th prompt (e.g. `/compare`) without coordination, breaking the D-13 "exactly 3 prompts" invariant | mitigate | Task 2 test 2 (`prompts/list returns exactly 3 prompts`) fails immediately if DASHBOARD_PROMPT_NAMES grows. The `DASHBOARD_PROMPT_NAMES.length === 3` assertion in Task 2 test 1 is a second guard. Additions require updating the test + CONTEXT.md D-13 together — catches the drift in code review. |
| T-30-05-08 | Information Disclosure | Prompt handler throws an error that leaks internal state | mitigate | Handlers are pure functions of `args` — no try/catch needed because there are no external calls (no DB, no network). The only throw path is if the SDK's zod validation catches an invalid argument shape, which it handles and returns as a standard JSON-RPC error. Our code contributes no additional error surface. |
