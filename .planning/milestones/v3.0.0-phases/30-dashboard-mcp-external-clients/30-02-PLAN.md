          }

          if (history.length === 0) {
            return errorEnvelope('No brand score recorded for this site in the current org');
          }
          const entry = history[0];
          return { content: [{ type: 'text', text: JSON.stringify({ siteUrl: args.siteUrl, computedAt: entry?.computedAt, score: entry?.result }, null, 2) }] };
        },
      );
    }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| MCP client → `/api/v1/mcp` | JSON-RPC tool-call args (LLM-generated, untrusted) cross into handlers |
| JWT claims → ToolContext.orgId | orgId from token drives every DB filter and cross-org guard in data tools |
| Tool args → scanService / storage | scanId and siteUrl are LLM-supplied; caller could guess cross-org values |
| `dashboard_scan_site` → external network | URL supplied by caller triggers real HTTP fetch against the target site |

| Threat ID | Category | Component | Disposition | Mitigation Plan |
