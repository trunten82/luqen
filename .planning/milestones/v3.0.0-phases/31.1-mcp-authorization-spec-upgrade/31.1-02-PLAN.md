Step 1. Create `packages/dashboard/src/auth/oauth-key-bootstrap.ts`:

```typescript
import { generateKeyPairSync, randomBytes } from 'node:crypto';
import { encryptSecret } from '../plugins/crypto.js';
import type { StorageAdapter } from '../db/adapter.js';

export async function ensureInitialSigningKey(
  storage: StorageAdapter,
  encryptionKey: string,
): Promise<void> {
  const active = await storage.oauthSigningKeys.listActiveKeys();
  if (active.length > 0) return;

  const { publicKey, privateKey } = generateKeyPairSync('rsa', {
    modulusLength: 2048,
    publicKeyEncoding: { type: 'spki', format: 'pem' },
    privateKeyEncoding: { type: 'pkcs8', format: 'pem' },
  });

  const kid = `k_${randomBytes(8).toString('hex')}`;
  const encryptedPrivate = encryptSecret(privateKey, encryptionKey);

  await storage.oauthSigningKeys.insertKey({
    kid,
    publicKeyPem: publicKey,
    encryptedPrivateKeyPem: encryptedPrivate,
  });
}
```

```typescript
import { createHash } from 'node:crypto';

export function verifyS256Challenge(codeVerifier: string, storedChallenge: string): boolean {
  if (codeVerifier.length < 43 || codeVerifier.length > 128) return false;
  const computed = createHash('sha256')
    .update(codeVerifier)
    .digest('base64url'); // Node 15+
  return computed === storedChallenge;
}
```

Step 3. Create `packages/dashboard/src/auth/oauth-signer.ts`:

```typescript
import { importPKCS8, SignJWT, type JWTPayload } from 'jose';
import { decryptSecret } from '../plugins/crypto.js';
import type { StorageAdapter } from '../db/adapter.js';
import type { SigningKey } from '../db/interfaces/oauth-signing-key-repository.js';

export interface MintAccessTokenInput {
  readonly sub: string;
  readonly scopes: readonly string[];
  readonly orgId: string;
  readonly aud: readonly string[];
  readonly role?: string;
  readonly expiresInSeconds: number; // 3600 = 1h per D-28
}

export interface DashboardSigner {
  readonly currentKid: string;
  mintAccessToken(input: MintAccessTokenInput): Promise<string>;
}

export async function createDashboardSigner(
  storage: StorageAdapter,
  encryptionKey: string,
): Promise<DashboardSigner> {
  const [activeKey] = await storage.oauthSigningKeys.listActiveKeys();
  if (activeKey === undefined) {
    throw new Error('No active OAuth signing key — run ensureInitialSigningKey first');
  }
  const privateKeyPem = decryptSecret(activeKey.encryptedPrivateKeyPem, encryptionKey);
  const privateKey = await importPKCS8(privateKeyPem, 'RS256');

  return {
    currentKid: activeKey.kid,
    async mintAccessToken(input: MintAccessTokenInput): Promise<string> {
      const claims: JWTPayload = {
        sub: input.sub,
        scopes: input.scopes,
        orgId: input.orgId,
        aud: input.aud,
        ...(input.role != null ? { role: input.role } : {}),
      };
      return new SignJWT(claims)
        .setProtectedHeader({ alg: 'RS256', kid: activeKey.kid })
        .setIssuedAt()
        .setExpirationTime(`${input.expiresInSeconds}s`)
        .setIssuer(getIssuer())
        .sign(privateKey);
    },
  };
}

function getIssuer(): string {
  // Must equal the issuer advertised in /.well-known/oauth-authorization-server (Task 3)
  return process.env['DASHBOARD_PUBLIC_URL'] ?? 'https://dashboard.luqen.local';
}
```

Step 4. Create `packages/dashboard/src/routes/oauth/jwks.ts`:

```typescript
import type { FastifyInstance } from 'fastify';
import { createPublicKey } from 'node:crypto';
import type { StorageAdapter } from '../../db/adapter.js';

export async function registerJwksRoutes(
  server: FastifyInstance,
  storage: StorageAdapter,
): Promise<void> {
  const handler = async (_req: unknown, reply: { header: (k: string, v: string) => unknown; send: (b: unknown) => unknown }) => {
    const keys = await storage.oauthSigningKeys.listPublishableKeys();
    const jwks = {
      keys: keys.map((k) => {
        const publicKeyObject = createPublicKey({ key: k.publicKeyPem, format: 'pem' });
        const jwk = publicKeyObject.export({ format: 'jwk' }) as { n: string; e: string };
        return {
          kid: k.kid,
          kty: 'RSA',
          alg: 'RS256',
          use: 'sig',
          n: jwk.n,
          e: jwk.e,
        };
      }),
    };
    reply.header('Cache-Control', 'public, max-age=3600');
    reply.header('Content-Type', 'application/json');
    return reply.send(jwks);
  };

  server.get('/oauth/jwks.json', handler as never);
  server.get('/.well-known/jwks.json', handler as never);
}
```

```typescript
  write: 'Trigger scans and make writes such as creating guidelines or running rescans',
  'admin.system': 'Administer the Luqen system globally',
  'admin.org': "Administer your active org's settings and users",
  'admin.users': "Manage users within your active org",

export async function registerAuthorizeRoutes(
  server: FastifyInstance,
  storage: StorageAdapter,

  server.get('/oauth/authorize', { preHandler: requireAuth }, async (req, reply) => {
    const q = req.query as Record<string, string | undefined>;

    // 1. Validate response_type
    if (q.response_type !== 'code') return reply.status(400).send({ error: 'unsupported_response_type' });
    // 2. Validate PKCE
    if (q.code_challenge_method !== 'S256') {
      return reply.status(400).send({ error: 'invalid_request', error_description: 'code_challenge_method must be S256' });
    }
    if (q.code_challenge == null || q.code_challenge.length < 43) {
      return reply.status(400).send({ error: 'invalid_request', error_description: 'code_challenge required' });
    }
    // 3. Validate scope whitelist
    const requestedScopes = (q.scope ?? 'read').split(/\s+/).filter((s) => s.length > 0);
    for (const s of requestedScopes) {
      if (!VALID_SCOPES.has(s)) return reply.status(400).send({ error: 'invalid_scope' });
    }
    // 4. Lookup client + validate redirect_uri
    if (q.client_id == null) return reply.status(400).send({ error: 'invalid_request', error_description: 'client_id required' });
    const client = await storage.oauthClients.findByClientId(q.client_id);
    if (client === null || client.revokedAt !== null) {
      return reply.status(400).send({ error: 'invalid_client' });
    }
    if (q.redirect_uri == null || !client.redirectUris.includes(q.redirect_uri)) {
      return reply.status(400).send({ error: 'invalid_request', error_description: 'redirect_uri mismatch' });
    }
    // 5. Validate resource (RFC 8707) — must be non-empty for MCPAUTH-01 aud binding
    const requestedResources = (q.resource ?? '').split(/\s+/).filter((s) => s.length > 0);
    if (requestedResources.length === 0) {
      return reply.status(400).send({ error: 'invalid_target', error_description: 'at least one resource parameter required' });
    }

    // 6. D-09 — admin.system scope requires admin.system permission
    const user = req.user!;
    const orgId = (req.session.orgId as string) ?? 'system';
    const perms = await resolveEffectivePermissions(storage.roles, user.id, user.role, orgId);
    if (requestedScopes.includes('admin.system') && !perms.has('admin.system')) {
      return reply.view('oauth-consent', {
        adminScopeBlocked: true,
        client,
        user,
      });
    }

    // 7. Consent coverage (D-20)
    const coverage = await storage.oauthConsents.checkCoverage({
      userId: user.id,
      clientId: client.clientId,
      requestedScopes,
      requestedResources,
    });

    if (coverage.covered) {
      // Fast path: mint code + redirect
      const code = randomBytes(32).toString('base64url');
      const expiresAt = new Date(Date.now() + 60_000).toISOString();
      await storage.oauthCodes.createCode({
        code,
        clientId: client.clientId,
        userId: user.id,
        redirectUri: q.redirect_uri,
        scope: requestedScopes.join(' '),
        resource: requestedResources.join(' '),
        codeChallenge: q.code_challenge,
        codeChallengeMethod: 'S256',
        orgId,
        expiresAt,
      });
      const redirectUrl = new URL(q.redirect_uri);
      redirectUrl.searchParams.set('code', code);
      if (q.state != null) redirectUrl.searchParams.set('state', q.state);
      return reply.redirect(redirectUrl.toString(), 302);
    }

    // Slow path: render consent screen
    return reply.view('oauth-consent', {
      adminScopeBlocked: false,
      client,
      user,
      orgName: orgId,
      clientId: client.clientId,
      redirectUri: q.redirect_uri,
      requestedScope: requestedScopes.join(' '),
      requestedResource: requestedResources.join(' '),
      resources: requestedResources,
      scopeDescriptions: requestedScopes.map((s) => ({ scope: s, description: SCOPE_DESCRIPTIONS[s] ?? s })),
      state: q.state ?? '',
      codeChallenge: q.code_challenge,
      csrfToken: req.session.csrfToken,
    });
  });

  server.post('/oauth/authorize/consent', { preHandler: requireAuth }, async (req, reply) => {
    // Order of checks (DEFENSE IN DEPTH — see T-31.1-02-03 + checker warning W2):
    //   1. CSRF token verification (@fastify/csrf-protection or session.csrfToken compare → 403 if missing/mismatch)
    //   2. Re-validate `body.redirect_uri` against `client.redirectUris[]` as the VERY FIRST step after CSRF.
    //      Mismatch → 400 `{error:'invalid_request', error_description:'redirect_uri mismatch'}` and RETURN IMMEDIATELY
```

```typescript
import type { FastifyInstance, FastifyRequest, FastifyReply } from 'fastify';
import { randomBytes, createHash } from 'node:crypto';
import type { StorageAdapter } from '../../db/adapter.js';
import { verifyS256Challenge } from '../../auth/oauth-pkce.js';
import type { DashboardSigner } from '../../auth/oauth-signer.js';

export async function registerTokenRoutes(
  server: FastifyInstance,
  storage: StorageAdapter,
  signer: DashboardSigner,
): Promise<void> {

  server.post('/oauth/token', async (req, reply) => {
    const body = req.body as Record<string, string | undefined>;
    // Parse client auth from body OR Authorization: Basic
    const { clientId, clientSecret } = extractClientCredentials(req, body);

    const client = clientId != null ? await storage.oauthClients.findByClientId(clientId) : null;
    if (client === null || client.revokedAt !== null) {
      return reply.status(400).send({ error: 'invalid_client' });
    }

    // Verify client authentication appropriate to token_endpoint_auth_method
    if (client.tokenEndpointAuthMethod === 'client_secret_basic') {
      if (clientSecret == null || !(await storage.oauthClients.verifyClientSecret(client.clientId, clientSecret))) {
        return reply.status(400).send({ error: 'invalid_client' });
      }
    } else {
      // public client — must NOT present a secret
      if (clientSecret != null && clientSecret.length > 0) {
        return reply.status(400).send({ error: 'invalid_client' });
      }
    }

    switch (body.grant_type) {
      case 'authorization_code':
        return handleAuthCodeGrant(storage, signer, client, body, reply);
      case 'refresh_token':
        return handleRefreshGrant(storage, signer, client, body, reply);
      case 'client_credentials':
        return handleClientCredentialsGrant(signer, client, body, reply);
      default:
        return reply.status(400).send({ error: 'unsupported_grant_type' });
    }
  });
}

async function handleAuthCodeGrant(storage, signer, client, body, reply) {
  if (body.code == null || body.code_verifier == null || body.redirect_uri == null) {
    return reply.status(400).send({ error: 'invalid_request' });
  }
  const row = await storage.oauthCodes.findAndConsume(body.code);
  if (row === null) return reply.status(400).send({ error: 'invalid_grant' });
  if (row.clientId !== client.clientId) return reply.status(400).send({ error: 'invalid_grant' });
  if (row.redirectUri !== body.redirect_uri) return reply.status(400).send({ error: 'invalid_grant' });
  if (!verifyS256Challenge(body.code_verifier, row.codeChallenge)) {
    return reply.status(400).send({ error: 'invalid_grant' });
  }
  // Mint tokens
  const scopes = row.scope.split(/\s+/).filter(Boolean);
  const resources = row.resource.split(/\s+/).filter(Boolean);
  const accessToken = await signer.mintAccessToken({
    sub: row.userId, scopes, orgId: row.orgId, aud: resources, expiresInSeconds: 3600,
  });
  const rawRefresh = randomBytes(32).toString('base64url');
  const refreshHash = createHash('sha256').update(rawRefresh).digest('hex');
  await storage.oauthRefresh.mint({
    tokenHash: refreshHash, clientId: client.clientId, userId: row.userId, orgId: row.orgId,
    scope: row.scope, resource: row.resource, parentId: null,
    absoluteExpiresAt: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000).toISOString(),
  });
  return reply.send({
    access_token: accessToken,
    token_type: 'Bearer',
    expires_in: 3600,
    refresh_token: rawRefresh,
    scope: row.scope,
  });
}

async function handleRefreshGrant(storage, signer, client, body, reply) {
  if (body.refresh_token == null) return reply.status(400).send({ error: 'invalid_request' });
  const presentedHash = createHash('sha256').update(body.refresh_token).digest('hex');
  const newRaw = randomBytes(32).toString('base64url');
  const newHash = createHash('sha256').update(newRaw).digest('hex');
  const result = await storage.oauthRefresh.rotate(presentedHash, newHash);

  if (result.kind === 'success') {
    if (result.parent.clientId !== client.clientId) {
      return reply.status(400).send({ error: 'invalid_grant' });
    }
    const scopes = result.child.scope.split(/\s+/).filter(Boolean);
    const resources = result.child.resource.split(/\s+/).filter(Boolean);
    const accessToken = await signer.mintAccessToken({
      sub: result.child.userId, scopes, orgId: result.child.orgId, aud: resources, expiresInSeconds: 3600,
    });
    return reply.send({
      access_token: accessToken,
      token_type: 'Bearer',
      expires_in: 3600,
      refresh_token: newRaw,
      scope: result.child.scope,
    });
  }
  // reuse_detected / expired / not_found → invalid_grant
  // Task 4 Plan 04 writes agent_audit_log entry when kind==='reuse_detected'
  return reply.status(400).send({ error: 'invalid_grant' });
}

async function handleClientCredentialsGrant(signer, client, body, reply) {
  const scopes = (body.scope ?? client.scope).split(/\s+/).filter(Boolean);
  const resources = (body.resource ?? '').split(/\s+/).filter(Boolean);
  // client_credentials on dashboard: sub = clientId, orgId = 'system'
  const accessToken = await signer.mintAccessToken({
    sub: client.clientId, scopes, orgId: 'system', aud: resources, expiresInSeconds: 3600,
  });
  return reply.send({
    access_token: accessToken,
    token_type: 'Bearer',
    expires_in: 3600,
    scope: scopes.join(' '),
  });
}
```

```typescript
import type { FastifyInstance, FastifyRequest, FastifyReply } from 'fastify';
import rateLimit from '@fastify/rate-limit';
import type { StorageAdapter } from '../../db/adapter.js';

const SUPPORTED_GRANTS = new Set(['authorization_code', 'refresh_token']);
const SUPPORTED_AUTH_METHODS = new Set(['none', 'client_secret_basic']);

function validateRedirectUri(uri: string): boolean {
  try {
    const u = new URL(uri);
    if (u.protocol === 'https:') return true;
    if (u.protocol === 'http:' && (u.hostname === 'localhost' || u.hostname === '127.0.0.1')) return true;
    return false;
  } catch {
    return false;
  }
}

export async function registerRegisterRoutes(
  server: FastifyInstance,
  storage: StorageAdapter,
): Promise<void> {

  // Scoped rate limit per D-17
  await server.register(async (instance) => {
    await instance.register(rateLimit, {
      max: 10,
      timeWindow: '1 hour',
      keyGenerator: (req) => req.ip,
      // onSend JSON hook — feedback_rate_limiter.md invariant
      addHeaders: { 'retry-after': true, 'x-ratelimit-limit': true, 'x-ratelimit-remaining': true },
      errorResponseBuilder: () => ({ error: 'too_many_requests', statusCode: 429 }),
    });

    instance.post('/oauth/register', async (req, reply) => {
      const body = (req.body ?? {}) as Record<string, unknown>;

      if (typeof body.client_name !== 'string' || body.client_name.length === 0) {
        return reply.status(400).send({ error: 'invalid_client_metadata', error_description: 'client_name required' });
      }
      if (!Array.isArray(body.redirect_uris) || body.redirect_uris.length === 0) {
        return reply.status(400).send({ error: 'invalid_redirect_uri' });
      }
      for (const uri of body.redirect_uris) {
        if (typeof uri !== 'string' || !validateRedirectUri(uri)) {
          return reply.status(400).send({ error: 'invalid_redirect_uri' });
        }
      }
      const grantTypes = Array.isArray(body.grant_types) ? body.grant_types : ['authorization_code', 'refresh_token'];
      for (const g of grantTypes) {
        if (typeof g !== 'string' || !SUPPORTED_GRANTS.has(g)) {
          return reply.status(400).send({ error: 'invalid_client_metadata', error_description: `unsupported grant_type: ${String(g)}` });
        }
      }
      const authMethod = typeof body.token_endpoint_auth_method === 'string' ? body.token_endpoint_auth_method : 'none';
      if (!SUPPORTED_AUTH_METHODS.has(authMethod)) {
        return reply.status(400).send({ error: 'invalid_client_metadata', error_description: 'unsupported token_endpoint_auth_method' });
      }

      const result = await storage.oauthClients.register({
        clientName: body.client_name,
        redirectUris: body.redirect_uris as string[],
        grantTypes: grantTypes as string[],
        tokenEndpointAuthMethod: authMethod as 'none' | 'client_secret_basic',
        scope: typeof body.scope === 'string' ? body.scope : 'read',
        softwareId: typeof body.software_id === 'string' ? body.software_id : undefined,
        softwareVersion: typeof body.software_version === 'string' ? body.software_version : undefined,
        registeredByUserId: undefined, // open DCR, no user binding unless admin-gated
      });

      return reply.status(201).send({
        client_id: result.clientId,
        client_id_issued_at: Math.floor(Date.parse(result.createdAt) / 1000),
        client_secret: result.clientSecret,
        token_endpoint_auth_method: authMethod,
        grant_types: grantTypes,
        redirect_uris: body.redirect_uris,
        client_name: body.client_name,
        scope: typeof body.scope === 'string' ? body.scope : 'read',
      });
    });
  });
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| External client → /oauth/register | Open DCR — untrusted body lands in oauth_clients_v2 |
| External client → /oauth/authorize | Untrusted query params (client_id, redirect_uri, scope, resource, code_challenge, state) |
| External client → /oauth/token | Untrusted body + potentially HTTP Basic auth |
| Browser user → POST /oauth/authorize/consent | CSRF-sensitive endpoint (cookie session auth + state-changing) |
| Filesystem → server process | Encrypted private signing key decrypted only in signer closure at mint time |
