<objective>
Create the shared MCP HTTP plugin foundation in @luqen/core that all Luqen services will reuse to expose their McpServer over Streamable HTTP with OAuth2 JWT authentication, RBAC tool filtering, and mandatory org isolation via JWT claims.

    ```typescript
    interface TokenPayloadLike {
      readonly sub: string;
      readonly scopes: readonly string[];
      readonly orgId?: string;

    export function extractToolContext(request: FastifyRequest): ToolContext {
      const payload = (request as FastifyRequest & { tokenPayload?: TokenPayloadLike }).tokenPayload;
      if (payload == null) {
        throw new Error('MCP request reached tool dispatch without authenticated tokenPayload — check preHandler order');
      }
      const orgId = (request as FastifyRequest & { orgId?: string }).orgId ?? 'system';
      const authType = (request as FastifyRequest & { authType?: string }).authType;
      const permissions = (request as FastifyRequest & { permissions?: Set<string> }).permissions ?? new Set<string>();
      return {
        orgId,
        userId: payload.sub,
        scopes: [...payload.scopes],
        permissions,
        authType: authType === 'apikey' ? 'apikey' : 'jwt',
      };
    }
    ```

    ```typescript
    import type { ToolMetadata } from './types.js';

    export function filterToolsByPermissions(
      allTools: readonly ToolMetadata[],
      effectivePerms: ReadonlySet<string>,
    ): readonly string[] {
      return allTools
        .filter((t) => t.requiredPermission == null || effectivePerms.has(t.requiredPermission))
        .map((t) => t.name);
    }

    export function filterToolsByScope(
      allTools: readonly ToolMetadata[],
      tokenScopes: readonly string[],
    ): readonly string[] {
      const hasAdmin = tokenScopes.includes('admin');
      const hasWrite = tokenScopes.includes('write') || hasAdmin;
      const hasRead = tokenScopes.includes('read') || hasWrite;
      return allTools
        .filter((t) => {
          if (t.requiredPermission == null) return true;
          if (hasAdmin) return true;
          const perm = t.requiredPermission;
          // *.manage permissions require write+ scope; *.view and other read permissions require read+ scope.
          if (perm.endsWith('.manage') || perm.includes('.delete') || perm.endsWith('.system') || perm.endsWith('.org')) {
            return hasWrite;
          }
          return hasRead;
        })
        .map((t) => t.name);
    }
    ```

    2) Create `packages/core/src/mcp/http-plugin.ts` implementing createMcpHttpPlugin with the COMMITTED SDK request-handler approach (blocker 4 fix):

    ```typescript
    import { AsyncLocalStorage } from 'node:async_hooks';
    import type { FastifyInstance, FastifyPluginAsync, FastifyRequest, FastifyReply } from 'fastify';
    import { McpServer } from '@modelcontextprotocol/sdk/server/mcp.js';
    import { StreamableHTTPServerTransport } from '@modelcontextprotocol/sdk/server/streamableHttp.js';
    import { ListToolsRequestSchema } from '@modelcontextprotocol/sdk/types.js';
    import { extractToolContext } from './auth.js';
    import { filterToolsByPermissions, filterToolsByScope } from './tool-filter.js';
    import type { ToolContext, ToolMetadata } from './types.js';

    export interface McpHttpPluginOptions {
      readonly mcpServer: McpServer;
      readonly toolMetadata: readonly ToolMetadata[];
      readonly requiredScope?: 'read' | 'write' | 'admin';
      readonly path?: string;
    }

    const SCOPE_HIERARCHY: Record<string, readonly string[]> = {
      admin: ['read', 'write', 'admin'],
      write: ['read', 'write'],
      read: ['read'],
    };

    function scopeCovers(tokenScopes: readonly string[], required: 'read' | 'write' | 'admin'): boolean {
      for (const s of tokenScopes) {
        if (SCOPE_HIERARCHY[s]?.includes(required)) return true;
      }
      return false;
    }

    const toolContextStore = new AsyncLocalStorage<ToolContext>();

    export function getCurrentToolContext(): ToolContext | undefined {
      return toolContextStore.getStore();
    }

    export async function createMcpHttpPlugin(
      options: McpHttpPluginOptions,
    ): Promise<FastifyPluginAsync> {
      const { mcpServer, toolMetadata, requiredScope = 'read', path = '/api/v1/mcp' } = options;

      // COMMITTED APPROACH (blocker 4 fix): register a tools/list handler on the
      // McpServer's underlying Server. The SDK's McpServer exposes `.server` which
      // is the low-level Server instance with setRequestHandler.
      //
      // This intercepts BEFORE the default tools/list handler, so we can filter
      // the manifest by the current caller's RBAC permissions (or scope for
      // service-to-service callers without RBAC).
      //
      // Tool metadata is captured in plugin closure — we look up the full tool
      // definitions (description, inputSchema) from the McpServer's registered
      // tool map so clients still receive full tool descriptors, just a subset.
      mcpServer.server.setRequestHandler(ListToolsRequestSchema, async () => {
        const ctx = getCurrentToolContext();
        const allowedNames: readonly string[] = ctx == null
          ? []  // no context = pre-auth probe; return empty list
          : ctx.permissions.size > 0
            ? filterToolsByPermissions(toolMetadata, ctx.permissions)
            : filterToolsByScope(toolMetadata, ctx.scopes);
        const allowedSet = new Set(allowedNames);

        // Look up the full registered tool list from McpServer's internal tool
        // registry. The SDK keeps these on `mcpServer._registeredTools` (private
        // but accessible). If the private field is missing on a future SDK,
        // fall back to emitting just the name + empty schema (LLM clients
        // still need a valid tools array shape).
        const registered = (mcpServer as unknown as {
          _registeredTools?: Record<string, { description?: string; inputSchema?: unknown }>;
        })._registeredTools ?? {};

        const tools = Object.entries(registered)
          .filter(([name]) => allowedSet.has(name))
          .map(([name, def]) => ({
            name,
            description: def.description ?? '',
            // Convert zod schema to JSON schema via the SDK's built-in path
            // (the SDK normally does this in its default tools/list handler).
            // If zodToJsonSchema is not reachable from here, pass the raw
            // inputSchema — the SDK-side consumer tolerates both shapes in 1.27.x.
            inputSchema: def.inputSchema ?? { type: 'object' },
          }));

        return { tools };
      });

      return async function mcpHttpPlugin(app: FastifyInstance): Promise<void> {
        app.post(path, async (request: FastifyRequest, reply: FastifyReply) => {
          let context: ToolContext;
          try {
            context = extractToolContext(request);
          } catch {
            return reply.status(401).send({ error: 'Not authenticated', statusCode: 401 });
          }
          if (!scopeCovers(context.scopes, requiredScope)) {
            return reply.status(403).send({
              error: `Insufficient scope. Required: ${requiredScope}`,
              statusCode: 403,
            });
          }
          const transport = new StreamableHTTPServerTransport({ sessionIdGenerator: undefined });
          reply.raw.on('close', () => {
            void transport.close();
          });
          await toolContextStore.run(context, async () => {
            await mcpServer.connect(transport);
            await transport.handleRequest(request.raw, reply.raw, request.body);
          });
        });
      };
    }
    ```
