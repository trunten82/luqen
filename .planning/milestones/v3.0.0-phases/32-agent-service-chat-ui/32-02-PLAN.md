         ```typescript
         export interface AgentConversationInput {
           readonly orgId: string;
           readonly userId: string;
           readonly messages: ReadonlyArray<ChatMessage>;
           readonly tools: ReadonlyArray<ToolDef>;
           readonly agentDisplayName: string;
           readonly signal?: AbortSignal;
         }

         export async function* executeAgentConversation(
           db: DbAdapter,
           adapterFactory: (type: string) => LLMProviderAdapter,
           input: AgentConversationInput,
         ): AsyncIterable<StreamFrame> {
           const models = await db.getModelsForCapability('agent-conversation', input.orgId);
           if (models.length === 0) throw new CapabilityNotConfiguredError('agent-conversation');

           // Resolve system prompt (org override OR default) and interpolate display name
           const override = await db.getPromptOverride('agent-system', input.orgId);
           const rawTemplate = override ?? buildAgentSystemPrompt();
           const systemContent = rawTemplate.replaceAll('{agentDisplayName}', input.agentDisplayName);

           const fullMessages: ChatMessage[] = [
             { role: 'system', content: systemContent },
             ...input.messages,
           ];

           const errors: Error[] = [];
           for (const model of models) {
             const provider = await db.getProvider(model.providerId);
             if (!provider) continue;
             const adapter = adapterFactory(provider.type);
             try {
               await adapter.connect({ baseUrl: provider.baseUrl, apiKey: provider.apiKey });
               if (typeof adapter.completeStream !== 'function') {
                 throw new Error(`Provider ${provider.type} does not support streaming`);
               }
               const iter = adapter.completeStream(
                 fullMessages,
                 { model: model.modelId, temperature: 0.3, maxTokens: 2048, timeout: provider.timeout ?? 30, tools: input.tools },
                 input.signal,
               );
               // Forward every frame; mid-stream errors are caller's concern (D-23)
               for await (const frame of iter) {
                 yield frame;
               }
               return;  // stream completed (or mid-stream error frame was yielded); DO NOT retry next provider
             } catch (err) {
               // Stream-open failure — try next provider
               errors.push(err instanceof Error ? err : new Error(String(err)));
               try { await adapter.disconnect(); } catch { /* ignore */ }
             }
           }
           throw new CapabilityExhaustedError('agent-conversation', models.length, errors[errors.length - 1]);
         }
         ```
