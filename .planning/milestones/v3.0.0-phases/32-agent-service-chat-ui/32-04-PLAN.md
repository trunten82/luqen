<!-- SSE frame schema (AI-SPEC §4b.1 verbatim) -->
```typescript
import { z } from 'zod';
export const TokenFrameSchema = z.object({ type: z.literal('token'), text: z.string() });
export const ToolCallsFrameSchema = z.object({
  type: z.literal('tool_calls'),
  calls: z.array(z.object({ id: z.string(), name: z.string(), args: z.record(z.unknown()) })),
});
export const PendingConfirmationFrameSchema = z.object({
  type: z.literal('pending_confirmation'),
  messageId: z.string(),
  toolName: z.string(),
  args: z.record(z.unknown()),
  confirmationText: z.string().optional(),   // from ToolMetadata.confirmationTemplate
});
export const DoneFrameSchema = z.object({ type: z.literal('done') });
export const ErrorFrameSchema = z.object({
  type: z.literal('error'),
  code: z.enum(['provider_failed', 'iteration_cap', 'rate_limited', 'internal', 'tool_timeout']),
  message: z.string(),
  retryable: z.boolean(),
});
export const SseFrameSchema = z.discriminatedUnion('type', [
  TokenFrameSchema, ToolCallsFrameSchema, PendingConfirmationFrameSchema, DoneFrameSchema, ErrorFrameSchema,
]);
export type SseFrame = z.infer<typeof SseFrameSchema>;
```

<!-- Route contract (D-20, D-21, D-22) -->
```
POST /agent/message — HTMX form POST, session-guarded
  - Body: { conversationId, content, _csrf }
  - Writes user message; returns 202 Accepted with the user message markup OOB (HTMX partial)
  - Kicks off AgentService.runTurn in the background OR returns a streamId the client uses to open SSE

GET /agent/stream/:conversationId — session-guarded; sets SSE headers; forwards AgentService emits
  - Response: text/event-stream with frames matching SseFrameSchema

POST /agent/confirm/:messageId — session-guarded; transitions pending_confirmation → sent; resumes loop
POST /agent/deny/:messageId — session-guarded; transitions pending_confirmation → denied; feeds "user declined" to next turn
GET /agent/panel — session-guarded; renders the drawer HTMX partial (used on drawer open)

All five routes rate-limited: max 60/min per user. Rate-limit JSON response is produced by a Fastify `onSend` hook on the `/agent/*` prefix that detects status 429, overrides `Content-Type` to `application/json`, and replaces the body with `{error:'rate_limited', retry_after_ms}`. The `onSend` hook is required because `@fastify/rate-limit`'s `errorResponseBuilder` is unreliable on this plugin version (memory: `feedback_rate_limiter.md`).
```
